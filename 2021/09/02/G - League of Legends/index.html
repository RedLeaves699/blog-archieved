    #include <cstdio>
#include <cstring>
#include <algorithm>
#include <functional>
#include <queue>

const int MAXN = 5e3 + 10;
struct interval {
    int l, r;
    bool operator <(const interval &i)const {
        return l == i.l ? r > i.r : l < i.l;
    }
}vt[MAXN], v[MAXN];

int len[MAXN], dp[MAXN][MAXN];
int main() {
    int n, K;
    scanf("%d%d", &n, &K);
    for (int i = 1; i <= n; ++i) {
        scanf("%d%d", &vt[i].l, &vt[i].r);
    }

    std::sort(vt + 1, vt + 1 + n);
    int cnt = 0, nn = 0, prev_r = 1e9;
    for (int i = n; i >= 1; --i) {
        if (prev_r <= vt[i].r) len[++cnt] = vt[i].r - vt[i].l;
        else prev_r = vt[i].r, v[++nn] = vt[i];
    }
    std::reverse(v + 1, v + 1 + nn);
    std::sort(len + 1, len + 1 + cnt, std::greater<int> ());

    std::fill_n(&dp[0][0], MAXN * MAXN, -1);
    dp[0][0] = 0;
    for (int i = 1; i <= K; ++i) {
        std::deque<int> dq;
        for (int j = 1; j <= nn; ++j) {
            if (~dp[i - 1][j - 1]) {
                while (!dq.empty() && dp[i - 1][dq.back()] + v[dq.back() + 1].r <= dp[i - 1][j - 1] + v[j].r) {
                    dq.pop_back();
                }
                dq.push_back(j - 1);
            }
            while (!dq.empty() && v[dq.front() + 1].r <= v[j].l) dq.pop_front();
            if (!dq.empty()) {
                dp[i][j] = dp[i - 1][dq.front()] + v[dq.front() + 1].r - v[j].l;
            }
        }
    }

    int sum = 0, ans = 0, tmp = std::min(K, cnt);
    for (int i = 0; i <= tmp; ++i) {
        sum += len[i];
        if(~dp[K - i][nn])ans = std::max(ans, sum + dp[K - i][nn]);
    }
    printf("%d\n", ans);
    return 0;
}

    YT had a magic permutation $a_{1},a_{2},\cdots, a_{n}$, and he constructed a sequence $b_{1},b_{2},\cdots b_{n}$ by the following pseudo code:

    Stk is an empty stack
for i = 1 to n :
    while ( Stk is not empty ) and ( Stk's top > a[i] ) :
        pop Stk
    push a[i]
    b[i]=Stk's size

    But he somehow forgot the permutation $a$, and only got some $k$ elements of $b_{i}$.

    Construct a permutation $a$ satisfying these $b_{i}$, or determine no such permutation exists.

    Here a permutation refers to a sequence that contains all integers from $1$ to ${n}$ exactly once.

std::queue <int> q;
std::vector <int> v;
void toposort(int n) {  
    for (int i = 1; i <= n; ++i) {
        if (in[i] == 0) q.push(i);
    }
    while (!q.empty()) {
        int t = q.front();q.pop();
        v.push_back(t);//æŒ‰ç…§æ‹“æ‰‘åºæ”¾ç½®ä½ç½®åº
        in[next[t]]--;

            q.push(next[t]);
        }
    }
    Print(v, n);
}

std::stack <int> s;
int solve() {
    int n, k;
    scanf("%d%d", &n, &k);
    for (int i = 1; i <= k; ++i) {
        int pos;
        scanf("%d", &pos);scanf("%d", &b[pos]);
    }
    for (int i = 1; i <= n; ++i) {
        int t;
        if (b[i]) {
            if (b[i] > i) return -1;
            while (s.size() && s.size() >= b[i]) t = s.top(), s.pop();
            next[t] = i;
        }
        if(!s.empty()) next[i] = s.top();
        s.push(i);
    }
    for (int i = 1; i <= n; ++i) {
        in[next[i]]++;
    }
    toposort(n);
    return 0;
}

    #include <stdio.h>
#include <string.h>

#define MAXN 3000000 + 100

struct num {
    int p, val;
}stack[MAXN];

void Print_array(int *a, int n) {
    for (int i = 1; i <= n; ++i) {
        if (i == 1)printf("%d", a[i]);
        if (i > 1)printf(" %d", a[i]);
    }
    printf("\n");
}

int ans[MAXN];
int solve() {
    int top = 0;
    int n, x;
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) {
        scanf("%d", &x);
        struct num xx = {i, x};
        if (top == 0) stack[++top] = xx, ans[stack[top].p] = 0;
        else {
            while (top && xx.val > stack[top].val) {
                ans[stack[top].p] = xx.p;
                --top;
            }
            stack[++top] = xx;
        }
    }
    while (top--) {
        ans[stack[top].p] = 0;
    }
    Print_array(ans, n);
    return 1;
}
int main() {
    solve();
    return 0;
}

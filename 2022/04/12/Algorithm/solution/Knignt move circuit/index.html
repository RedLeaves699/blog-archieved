    马的周游问题(knight move circuit)

    题目描述

    给定一个\(n\)行\(m\)列的棋盘，任意给出一个马的周游路线。周游路线是指从起始点开始，不重不漏的经过棋盘上的每一个格子，最后再回到起始点的路线。

    样例输入

    6 5

    样例输出

        2   25   22   13    4
   23   12    3   30   21
   26    1   24    5   14
   11   16    9   20   29
    8   27   18   15    6
   17   10    7   28   19

    解答

    本题是一个哈密顿路问题，正常来说，哈密顿问题是一个NP完全问题。不过，由马的周游问题的特殊性，我们可以通过将问题分解为子问题的方式。

    周游问题的特殊性见下图。

    从图中可以看到，我们将问题分解后，得到的子问题的解法大致相同，而且两个子问题很容易合并。

    子问题合并

    子问题合并的规则如下。令\(p_1, p_2\in E\)，\(p_1',p_2' \in E'\)：

    1. \(p_1, p_2\)之间，\(p_1',p_2'\)之间都有无向边连接。
    2. \(p_1, p'_1\)和\(p_2,p_2'\)可达，或\(p_1, p'_2\)和\(p_1', p_2\)可达。

    之后去除原来的边，加上连接两个图的边，即可完成两个回路的连接。

    程序中void merge(struct subq q1, struct subq q2)对应了此过程。

    子问题求解

    将要求解的子问题\(q\)具有如下的性质：

    1. 问题的范围可以通过四个变量描述，分别是子问题的起始位置\((x_0, y_0)\)，子问题的规模\((n, m)\)。
    2. 必须满足\(n * m\ \text{mod}\ 2 == 0, \min(n,m)\geq5\)，问题才有解1。
    3. 问题是可合并的。
    4. 当问题规模较小时，可以使用回溯法求解。

    回溯法的启发式优化：

    1. 扩展节点时，优先扩展可能选择数目最小的节点。代码中的int get_possible(int x, int y, struct subq q)函数实现了计算一个节点的可能选择数目。
    2. 扩展节点时，在(1)的条件下，尽可能扩展离中点远的节点。
    3. 起始节点选在中间。

    程序中void dfs(int x, int y, int cnt, struct subq q)，void solve(struct subq q)对应了此过程。

    原问题分解

    经过优化的回溯法非常适合求解\(|m-n|\leq 3(5\leq m,n \leq 16)\)的数据，所以在分解问题时，我们在保证分解后问题都有解的情况下，让子问题尽量接近一个正方形。

    我们有如下的分解方法：

    1. 如果\(n,m\)均为偶数，如果\(n\geq m\)，那么分解问题为两个\(n/2\)，\(m\)的子问题。\(n\leq m\)同理。
    2. 如果\(n\)是奇数，\(m\)是偶数，如果\(n\geq m\)，那么分解问题为\(\lfloor\frac{n}{2}\rfloor,m\)与\(n-\lfloor\frac{n}{2}\rfloor\)的子问题。如果\(n\leq m\)，令\(t\)为最接近\(\lfloor\frac{n}{2}\rfloor\)的偶数，则将问题分解为\(n, t\)与\(n, m - t\)的子问题。另一种情况同理。
    3. 如果\(n,m\)均为奇数，无解。

    程序实现中void div_conquer(struct subq q)函数对应了此过程。

    时间复杂度

    1. 分治法思想

      令\(n\times m\)的周游问题规模为\(N\)。每次合并子问题需要的代价是图的合并和寻找满足合并条件的点产生的总代价，为子问题的规模。列出复杂度的递推方程： \[ W(N)=W(\frac{N}{2})+N \] 通过主定理或换元迭代法，可以求出分治法思想的时间复杂度为\(O(N\log N)\)，也就是\(O(nm\log nm)\)

    2. 回溯法求解

      对于最终的小规模问题，由于在暴力求解中使用了启发式函数进行搜索，时间复杂度变得难以估计，理论上的时间复杂度是\(O(2^N)\)的，但实际情况要好的多。这里假设解决所有子问题的总复杂度为\(k\)，它是一个不能忽略的很大的常数，它是远大于\(\log N\)的，但又远小于\(2^N\)。

      通过单独运行回溯法程序，是可以在有限时间内求解100以内的问题的（大约30秒）。

    3. 结论

      总时间复杂度为： \[ O(nmk) \]

    程序实现

    下面的程序可以实现1s内对于\(m,n\leq 260\)的求解，可以在一分钟内求解\(m,n\leq 1000\)规模的问题。代码中的关键部分都添加了注释以说明各个模块的内容。

    /**
 * @file horse circuit problem.cpp
 * @author Redleaves699 (Algorithm_sort@163.com)
 * @brief A programme to solve horse circuit problem. 
 * @details compiler options: g++ horse.cpp -o horse
 * @version 1.0
 * @date 2022-04-12
 * @copyright Copyright (c) 2022
 * 
 */

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#include <vector>
#include <cassert>
#include <ctime>
#include <cstdlib>
#define MAXN (3000 + 10)
#define min(a, b) ((a) < (b) ? (a) : (b))
#define abs(a) ((a) > 0 ? (a) : (-(a)))

int book[MAXN][MAXN], book2[MAXN][MAXN]; // book在暴力求解中标记点是否访问过，book2表示最终结果
std::vector<int> G[MAXN * MAXN], G_[MAXN * MAXN]; // G_在暴力求解中存储子图，G表示最终结果图
// 方向数组，用来表示马所有可能的前进方向

int dir[][2] = { {-2, -1}, {-1, -2}, {1, -2}, {2, -1}, {2, 1}, {1, 2}, {-1, 2}, {-2, 1} };

int n, m; // 问题的规模
/**
 * @brief subq结构体表示问题的形式
 * x0, y0 : 子问题右上角的坐标
 * n, m : 子问题的规模
 * startx, starty : 子问题中马开始行动的位置。我们总是选择中间位置。
 */
struct subq{
    int x0, y0;
    int n, m;
    int startx, starty; 
};

/**
 * @brief 这个结构体是为了启发式搜索服务的，它描述了马在某个位置的可能选择
 * x, y : 马当前的位置
 * possible : 马在当前位置能够跳到点的数目
 * distance : 距离子问题中点的距离
 * dirw : 马能够选择的方向
 * overload < : for c++ stl priority_queue
 */
struct choice{
    int x, y, possible, distance, dirw;
    bool operator < (const struct choice &c) const{
        if (possible == c.possible) {
            if (distance == c.distance) {
                return dirw > c.dirw;
            }
            else return distance < c.distance;
        }
        else return possible > c.possible;
    }
};

/**
 * @brief 一个计算矩阵中点编号的内联函数
 * @param x coodinate x
 * @param y coodinate y
 * @return int 
 */
inline int num(int x, int y) {
    return m * x + y;
}

/**
 * @brief 判断一个点是否超过了子问题的范围
 * @param x coodinate x
 * @param y coodinate y
 * @param q 子问题信息
 * @return true/false
 */
inline bool check_in_q(int x, int y, struct subq q) {
    return q.x0 <= x && x < q.x0 + q.n && q.y0 <= y && y < q.y0 + q.m;
}

/**
 * @brief 获取马在某个位置所有可能的选择
 * @param x coodinate x
 * @param y coodinate y
 * @param q 子问题信息
 * @return int 
 */
int get_possible(int x, int y, struct subq q) {
    int cnt = 0;
    for (int i = 0; i < 8; ++i) {
        int x_ = x + dir[i][0], y_ = y + dir[i][1];
        if (check_in_q(x_, y_, q) && !book[x_][y_]) {
            ++cnt;
        }
    }
    return cnt;
}

/**
 * @brief 计算一个点距离子问题中点的距离
 * @param x coodinate x
 * @param y coodinate y
 * @param q 子问题信息
 * @return int 
 */
int get_distance(int x, int y, struct subq q) {
    return abs(x - (2 * q.x0 + q.n) / 2) + abs(y - (2 * q.y0 + q.m) / 2);
}

/**
 * @brief 判断图中编号num1,num2的点是否可达
 * @param num1 we define that num1 as the number of point (num1 / m, num1 % m)
 * @param num2 we define that num2 as the number of point (num2 / m, num2 % m)
 * @return true/false 
 */
bool existPath(int num1, int num2) {
    for (int i = 0; i < G[num1].size(); ++i) {
        if (G[num1][i] == num2) return true;
    }
    for (int i = 0; i < G[num2].size(); ++i) {
        if (G[num2][i] == num1) return true;
    }
    return false;
}

/**
 * @brief 暴力寻找四个满足合并条件的点
 * @param q1 sub-question1
 * @param q2 sub-question2
 * @return struct subq (这里subq的意思不再是子问题，它仅仅是为这个函数返回多个值提供了便利)
 */
struct subq find_pairs(struct subq q1, struct subq q2) {
    struct subq ans = {-1, -1};
    int s1, s2;
    // 看看应该枚举什么地方
    if (q1.x0 == q2.x0) {
        s2 = q2.y0 - 1, s1 = q1.x0;
    }
    if (q1.y0 == q2.y0) {
        s1 = q2.x0 - 1, s2 = q1.y0;
    }
    // 暴力寻找四个点
    for (int x = s1; x < q1.x0 + q1.n; ++x) {
        for (int y = s2; y < q1.y0 + q1.m; ++y) {
            for (int i = 0; i < 8; ++i) {
                int x_ = x + dir[i][0], y_ = y + dir[i][1];
                if (check_in_q(x_, y_, q2)) {
                    for (int j = 0; j < 8; ++j) {
                        int x__ = x_ + dir[j][0], y__ = y_ + dir[j][1];
                        if (check_in_q(x__, y__, q2) && existPath(num(x_, y_), num(x__, y__))) {
                            for (int k = 0; k < 8; ++k) {
                                int x4 = x__ + dir[k][0], y4 = y__ + dir[k][1];
                                if (check_in_q(x4, y4, q1) && existPath(num(x, y), num(x4, y4))) {
                                    return ans = {x, y, i, j, k, 0};
                                }
                            }
                        }
                    }
                }
            }
        }
    }
    return ans;
}

bool note[MAXN * MAXN]; // 只被travel函数使用，用来标记某个点是否被访问
/**
 * @brief 遍历新生成的图，将生成的图重新编号
 * @param x coodinate x
 * @param y coodinate y
 * @param q1 子问题1的信息
 * @param q2 子问题2的信息
 */
void travel(int x, int y, struct subq q1, struct subq q2) {
    int xx = x, yy = y;
    memset(note, 0, sizeof(note));
    note[num(xx, yy)] = 1;
    book2[xx][yy] = 1;
    int n = q1.m * q1.n + q2.m * q2.n;
    for (int cnt = 2; cnt <= n; ++cnt) {
        for (int i = 0; i < 8; ++i) {
            int x_ = xx + dir[i][0], y_ = yy + dir[i][1];
            if ((check_in_q(x_, y_, q1) || check_in_q(x_, y_, q2)) && !note[num(x_, y_)] 
                && existPath(num(x_, y_), num(xx, yy))) {
                note[num(x_, y_)] = 1;
                book2[x_][y_] = cnt;
                xx = x_, yy = y_;
                break;
            }
        }
    }
}

int ok = 0; // 如果dfs寻找到答案，就不再寻找了
const int SmallestSize = 16; // the smallest scale need to be solved by dfs
/**
 * @brief 使用启发式搜索暴力解决数据规模为 5<= n,m <= SmallestSize(line : 197)的问题
 * @param x coodinate x
 * @param y coodinate y
 * @param cnt 被访问到的点的个数，用来标记点的访问次序
 * @param q 子问题信息
 */
void dfs(int x, int y, int cnt, struct subq q) {
    if (ok) return;
    if (cnt == q.m * q.n && !ok) {
        for (int i = 0; i < 8; ++i) {
            int x_ = x + dir[i][0], y_ = y + dir[i][1];
            // 检查并记录结果
            if (q.startx == x_ && q.starty == y_) {
                for (int i = q.x0; i < q.x0 + q.n; ++i) {
                    for (int j = q.y0; j < q.y0 + q.m; ++j) {
                        G[num(i, j)] = G_[num(i, j)];
                        book2[i][j] = book[i][j];
                    }
                }
                G[num(q.startx,q.starty)].push_back(num(x, y));
                G[num(x, y)].push_back(num(q.startx,q.starty));
                ok = 1;
                return;
            }
        }
        return;
    }
    // 使用stl的优先队列实现启发式搜索，方法在前面已经介绍
    std::priority_queue<choice> pq;
    for (int i = 0; i < 8; ++i) {
        int x_ = x + dir[i][0], y_ = y + dir[i][1];
        if (check_in_q(x_, y_, q) && !book[x_][y_] && ok == 0) {
            struct choice c = {x_, y_, get_possible(x_, y_, q), get_distance(x_, y_, q), i};
            pq.push(c);
        }
    }
    // 尝试搜索所有可能的情况
    while (!pq.empty()) {
        struct choice c = pq.top();pq.pop();
        book[c.x][c.y] = cnt + 1; // 标记已经被访问
        int t = num(x, y), t_ = num(c.x, c.y);
        G_[t].push_back(t_); // 建立图
        G_[t_].push_back(t);
        dfs(c.x, c.y, cnt + 1, q);
        // 恢复原状
        G_[t].pop_back(); 
        G_[t_].pop_back();
        book[c.x][c.y] = 0;
    }
}

/**
 * @brief 打印最终结果
 * @param q the information of the sub-question
 */
void Print_Ans(struct subq q) {
    for (int i = q.x0; i < q.x0 + q.n; ++i) {
        for (int j = q.y0; j < q.y0 + q.m; ++j) {
            printf("%6d", book2[i][j]);
        }
        puts("");
    }
}

/**
 * @brief 使用dfs解决5<= n,m <= SmallestSize(line : 202)子问题的前期准备工作
 * @param q 子问题信息
 */
void solve(struct subq q) {
    if (q.n * q.m % 2 == 1 || min(q.n, q.m) < 5) {
        printf("(%d, %d) No answer!\n", q.n, q.m);
        return ;
    }
    book[q.startx][q.starty] = 1;
    ok = 0;
    dfs(q.startx, q.starty, 1, q);
    book[q.startx][q.starty] = 0;
}

/**
 * @brief 删除stl vector中数值为val的元素
 * @param v a c++ stl vector
 * @param val 数值val.
 */
void rmv(std::vector<int> &v, int val) {
    for (auto i = v.begin(); i < v.end(); ++i) {
        if (*i == val) v.erase(i);
    }
}

/**
 * @brief 合并子问题q1,q2
 * @param q1 子问题q1的信息
 * @param q2 子问题q2的信息
 */
void merge(struct subq q1, struct subq q2) {
    struct subq ans = find_pairs(q1, q2);
    if (ans.x0 == -1) {
        //如果什么结果都没返回，证明程序出了问题
        printf("Assert!\n");
        printf("q1 = %d %d %d %d, q2 = %d %d %d %d\n", q1.x0, q1.y0, q1.n, q1.m, q2.x0, q2.y0, q2.n, q2.m);
        assert(ans.x0 == -1);
    }
    // 得到需要合并的四个点
    int x1 = ans.x0, y1 = ans.y0;
    int x2 = x1 + dir[ans.n][0], y2 = y1 + dir[ans.n][1];
    int x3 = x2 + dir[ans.m][0], y3 = y2 + dir[ans.m][1];
    int x4 = x3 + dir[ans.startx][0], y4 = y3 + dir[ans.startx][1];

    // 合并两个子图，建立新图
    rmv(G[num(x1, y1)], num(x4, y4));
    rmv(G[num(x4, y4)], num(x1, y1));
    rmv(G[num(x2, y2)], num(x3, y3));
    rmv(G[num(x3, y3)], num(x2, y2));
    G[num(x2, y2)].push_back(num(x1, y1));G[num(x1, y1)].push_back(num(x2, y2));
    G[num(x4, y4)].push_back(num(x3, y3));G[num(x3, y3)].push_back(num(x4, y4));
    // remark the new graph

}

/**
 * @brief 使用分治法解决子问题q，原理已经在上面详细说明
 * @param q 子问题q的信息
 */
void div_conquer(struct subq q) {
    // printf("q = %d %d\n", q.n, q.m); // for debug use : watch divided sub-questions
    if (5 <= q.n && q.n <= SmallestSize && 5 <= q.m && q.m <= SmallestSize) {
        solve(q);
        return;
    }
    struct subq q1, q2;
    if ((q.m % 2 == 0) && (q.n % 2 == 0)) {
        if (q.n >= q.m) {
            q1.x0 = q.x0, q1.y0 = q.y0, q1.n = q.n / 2, q1.m = q.m;
            q1.startx = (2 * q1.x0 + q1.n) / 2, q1.starty = (2 * q1.y0 + q1.m) / 2;
            q2.x0 = q.x0 + q.n / 2, q2.y0 = q.y0, q2.n = q.n / 2, q2.m = q.m;
            q2.startx = (2 * q2.x0 + q2.n) / 2, q2.starty = (2 * q2.y0 + q2.m) / 2;
        }
        else {
            q1.x0 = q.x0, q1.y0 = q.y0, q1.n = q.n, q1.m = q.m / 2;
            q1.startx = (2 * q1.x0 + q1.n) / 2, q1.starty = (2 * q1.y0 + q1.m) / 2;
            q2.x0 = q.x0, q2.y0 = q.y0 + q.m / 2, q2.n = q.n, q2.m = q.m / 2;
            q2.startx = (2 * q2.x0 + q2.n) / 2, q2.starty = (2 * q2.y0 + q2.m) / 2;
        }
    }
    else if ((q.m % 2 == 1) && (q.n % 2 == 0)) {
        if (q.n >= q.m) {
            int n_;
            if ((q.n / 2) % 2) {
                n_ = q.n / 2 + 1;
            }
            else n_ = q.n / 2;
            q1.x0 = q.x0, q1.y0 = q.y0, q1.n = n_, q1.m = q.m;
            q1.startx = (2 * q1.x0 + q1.n) / 2, q1.starty = (2 * q1.y0 + q1.m) / 2;
            q2.x0 = q.x0 + n_, q2.y0 = q.y0, q2.n = q.n - n_, q2.m = q.m;
            q2.startx = (2 * q2.x0 + q2.n) / 2, q2.starty = (2 * q2.y0 + q2.m) / 2;
        }
        else {
            int m_;
            if ((q.m / 2) % 2) {
                m_ = q.m / 2 + 1;
            }
            else m_ = q.m / 2;
            q1.x0 = q.x0, q1.y0 = q.y0, q1.n = q.n, q1.m = m_;
            q1.startx = (2 * q1.x0 + q1.n) / 2, q1.starty = (2 * q1.y0 + q1.m) / 2;
            q2.x0 = q.x0, q2.y0 = q.y0 + m_, q2.n = q.n, q2.m = q.m - m_;
            q2.startx = (2 * q2.x0 + q2.n) / 2, q2.starty = (2 * q2.y0 + q2.m) / 2;
        }
    }
    else if ((q.m % 2 == 0) && (q.n % 2 == 1)) {
        if (q.n >= q.m) {
            int n_;
            if ((q.n / 2) % 2) {
                n_ = q.n / 2 + 1;
            }
            else n_ = q.n / 2;
            q1.x0 = q.x0, q1.y0 = q.y0, q1.n = n_, q1.m = q.m;
            q1.startx = (2 * q1.x0 + q1.n) / 2, q1.starty = (2 * q1.y0 + q1.m) / 2;
            q2.x0 = q.x0 + n_, q2.y0 = q.y0, q2.n = q.n - n_, q2.m = q.m;
            q2.startx = (2 * q2.x0 + q2.n) / 2, q2.starty = (2 * q2.y0 + q2.m) / 2;
        }
        else {
            int m_;
            if ((q.m / 2) % 2) {
                m_ = q.m / 2 + 1;
            }
            else m_ = q.m / 2;
            q1.x0 = q.x0, q1.y0 = q.y0, q1.n = q.n, q1.m = m_;
            q1.startx = (2 * q1.x0 + q1.n) / 2, q1.starty = (2 * q1.y0 + q1.m) / 2;
            q2.x0 = q.x0, q2.y0 = q.y0 + m_, q2.n = q.n, q2.m = q.m - m_;
            q2.startx = (2 * q2.x0 + q2.n) / 2, q2.starty = (2 * q2.y0 + q2.m) / 2;
        }
    }
    else {
        printf("Runtime Error. InValid divide! (%d %d)\n", q.n, q.m);
        assert(0);
    }
    div_conquer(q1), div_conquer(q2);
    merge(q1, q2);
    travel(q.x0, q.y0, q1, q2);
}
